  let state = { checked: {}, notes: {}, images: {} };
  let allRecords = [];

  // --- เริ่มต้นระบบ ---
  window.init = function() {
    buildChecklist();
    listenToDatabase();
    document.getElementById('f-date').value = new Date().toISOString().split('T')[0];
  };

  // --- ระบบสลับหน้า ---
  window.showPage = function(pageId, btn) {
    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
    document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
    document.getElementById(pageId).classList.add('active');
    btn.classList.add('active');
  };

  // --- สร้างหน้าตา Checklist ---
  function buildChecklist() {
    const container = document.getElementById('checklist-container');
    container.innerHTML = SECTIONS.map(s => `
      <div class="checklist-section">
        <div class="section-header" onclick="this.nextElementSibling.classList.toggle('open')" style="border-left: 5px solid ${s.color}">
          <strong>${s.title}</strong>
          <span>คลิกเพื่อเปิด/ปิด</span>
        </div>
        <div class="section-items">
          ${s.items.map((item, i) => {
            const key = `${s.id}_${i}`;
            return `
              <div class="check-item" id="ci-${key}" style="flex-direction: column;">
                <div class="item-main">
                  <input type="checkbox" id="cb-${key}" onchange="updateCheck('${key}')">
                  <div style="flex:1"><label for="cb-${key}">${item}</label></div>
                  
                  <span id="loader-${key}" class="loading-spinner">กำลังอัปโหลด...</span>
                  <label class="upload-label" title="แนบรูปภาพ">
                    📷 <input type="file" style="display:none" accept="image/*" onchange="handleFileUpload(this, '${key}')">
                  </label>
                  
                  <button onclick="document.getElementById('ci-${key}').classList.toggle('show-note')" style="background:none;border:none;cursor:pointer;">📝</button>
                </div>
                <img id="img-${key}" class="img-preview">
                <textarea class="notes-field" id="note-${key}" onchange="state.notes['${key}']=this.value" placeholder="หมายเหตุ..."></textarea>
              </div>
            `;
          }).join('')}
        </div>
      </div>
    `).join('');
  }

  // --- ฟังก์ชันอัปโหลดรูปภาพ ---
  window.handleFileUpload = async function(input, key) {
    const file = input.files[0];
    if (!file) return;
    const id = document.getElementById('f-id').value;
    if (!id) return alert("กรุณาใส่เลขบัตรประชาชนก่อนอัปโหลดรูป เพื่อใช้เป็นรหัสอ้างอิง");

    const loader = document.getElementById(`loader-${key}`);
    const preview = document.getElementById(`img-${key}`);
    
    try {
      loader.style.display = 'inline';
      const fileRef = ref(storage, `warrants/${id}/${key}_${file.name}`);
      await uploadBytes(fileRef, file);
      const url = await getDownloadURL(fileRef);
      
      state.images[key] = url;
      preview.src = url;
      preview.classList.add('active');
      showToast("อัปโหลดรูปสำเร็จ ✓");
    } catch (e) {
      alert("อัปโหลดพลาด: " + e.message);
    } finally {
      loader.style.display = 'none';
    }
  };

  window.updateCheck = (key) => { state.checked[key] = document.getElementById(`cb-${key}`).checked; };

  // --- ระบบดึงข้อมูลแบบ Real-time ---
  function listenToDatabase() {
    const q = query(collection(db, "warrants"), orderBy("profile.date", "desc"));
    onSnapshot(q, (snapshot) => {
      allRecords = [];
      const html = snapshot.docs.map(doc => {
        const data = doc.data();
        allRecords.push(data);
        const checkedCount = Object.values(data.checked || {}).filter(v => v).length;
        return `
          <tr onclick="loadRecord('${data.profile.id}')">
            <td><strong>${data.profile.id}</strong></td>
            <td>${data.profile.name}</td>
            <td>${data.profile.charge}</td>
            <td><span class="badge-status">ทำแล้ว ${checkedCount} รายการ</span></td>
            <td>${data.profile.date}</td>
          </tr>
        `;
      }).join('');
      document.getElementById('db-list').innerHTML = html || '<tr><td colspan="5" style="text-align:center;">ไม่พบข้อมูล</td></tr>';
    });
  }

  // ค้นหาในตาราง
  window.filterTable = function() {
    const val = document.getElementById('db-search').value.toLowerCase();
    const rows = document.querySelectorAll('#db-list tr');
    rows.forEach(row => {
      row.style.display = row.innerText.toLowerCase().includes(val) ? '' : 'none';
    });
  };

  // ดึงข้อมูลมาใส่ฟอร์ม
  window.loadRecord = function(id) {
    const data = allRecords.find(r => r.profile.id === id);
    if (!data) return;
    
    Object.entries(data.profile).forEach(([k, v]) => {
      const el = document.getElementById(`f-${k}`);
      if (el) el.value = v;
    });

    state.checked = data.checked || {};
    state.notes = data.notes || {};
    state.images = data.images || {};

    SECTIONS.forEach(s => {
      s.items.forEach((_, i) => {
        const key = `${s.id}_${i}`;
        const cb = document.getElementById(`cb-${key}`);
        const nt = document.getElementById(`note-${key}`);
        const img = document.getElementById(`img-${key}`);

        if(cb) cb.checked = state.checked[key] || false;
        if(nt) nt.value = state.notes[key] || '';
        
        if(img) {
          if(state.images[key]) {
            img.src = state.images[key];
            img.classList.add('active');
          } else {
            img.src = '';
            img.classList.remove('active');
          }
        }
      });
    });
    
    showToast("โหลดข้อมูลสำเร็จ");
    // สลับกลับไปหน้าฟอร์มกรอกข้อมูลอัตโนมัติ
    showPage('page-home', document.querySelectorAll('.nav-btn')[0]);
  };

  // --- ระบบบันทึก ---
  window.saveData = async function() {
    const id = document.getElementById('f-id').value;
    if (!id) return alert("กรุณากรอกเลขบัตรประชาชน");

    const docRef = doc(db, "warrants", id);
    const docSnap = await getDoc(docRef);
    if (docSnap.exists()) {
      if (!confirm("เลขบัตรนี้มีข้อมูลอยู่แล้ว คุณต้องการบันทึกอัปเดตทับข้อมูลเดิมใช่หรือไม่?")) return;
    }

    const data = {
      profile: {
        id: id,
        name: document.getElementById('f-name').value,
        gender: document.getElementById('f-gender').value,
        age: document.getElementById('f-age').value,
        warrant: document.getElementById('f-warrant').value,
        charge: document.getElementById('f-charge').value,
        address: document.getElementById('f-address').value,
        officer: document.getElementById('f-officer').value,
        date: document.getElementById('f-date').value
      },
      checked: state.checked,
      notes: state.notes,
      images: state.images
    };

    try {
      await setDoc(docRef, data);
      showToast("บันทึกข้อมูลลงฐานข้อมูลแล้ว ✓");
    } catch (e) { alert("Error: " + e.message); }
  };

  // --- ระบบ Export PPT ---
  window.exportToPPT = function() {
    let pptx = new PptxGenJS();
    let slide = pptx.addSlide();

    const name = document.getElementById('f-name').value || "ไม่ระบุชื่อ";

    slide.addText(`สรุปข้อมูลผู้ต้องหา: ${name}`, { x: 0.5, y: 0.5, fontSize: 24, color: 'C0392B', bold: true });
    
    let rows = [
      ["หัวข้อ", "รายละเอียด"],
      ["เลขบัตรประชาชน", document.getElementById('f-id').value],
      ["เลขหมายจับ", document.getElementById('f-warrant').value],
      ["ข้อหา", document.getElementById('f-charge').value],
      ["ผู้บันทึก", document.getElementById('f-officer').value]
    ];

    slide.addTable(rows, { x: 0.5, y: 1.2, w: 9, border: { pt: 1, color: 'BBBBBB' }, fontSize: 14 });
    
    const done = Object.values(state.checked).filter(v => v).length;
    slide.addText(`ความคืบหน้าการตรวจสอบ: ทำแล้ว ${done} รายการ`, { x: 0.5, y: 4.5, fontSize: 16 });

    for (const key in state.images) {
      if (state.images[key]) {
        let sImg = pptx.addSlide();
        const sectionId = key.split('_')[0];
        const itemIdx = key.split('_')[1];
        
        const sec = SECTIONS.find(s => s.id == sectionId);
        const itemTitle = sec ? sec.items[itemIdx] : "หลักฐานเพิ่มเติม";

        sImg.addText(`ภาพประกอบ: ${itemTitle}`, { x:0.5, y:0.3, fontSize:18, bold:true });
        sImg.addImage({ path: state.images[key], x:0.5, y:0.8, w:6, h:4 });
        
        if(state.notes[key]) {
          sImg.addText(`หมายเหตุ: ${state.notes[key]}`, { x:0.5, y:5.0, fontSize:12 });
        }
      }
    }
    pptx.writeFile({ fileName: `รายงาน_${name}.pptx` });
  };

  window.showToast = function(msg) {
    const t = document.getElementById('toast'); t.textContent = msg; t.classList.add('show');
    setTimeout(() => t.classList.remove('show'), 3000);
  };

  window.confirmClear = function() { if(confirm("ต้องการล้างข้อมูลในฟอร์มทั้งหมดใช่หรือไม่?")) location.reload(); };

  init();
</script>
